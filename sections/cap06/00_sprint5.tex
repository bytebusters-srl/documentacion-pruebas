\section{Alcance y Objetivos de V\&V del Sprint}
\label{sec:s5-alcance}

El Sprint 5 preparó la versión de lanzamiento. El objetivo de V\&V fue \textbf{validar la
preparación para producción}: exactitud de las métricas y su control de acceso, cobertura
de internacionalización, persistencia del tema, integridad de los \textit{seeders} y
\textit{backups}, y el correcto empaquetado monolítico con \textit{smoke tests}
post-despliegue.

\subsection{Funcionalidades y Componentes Evaluados}
\label{sec:s5-funcionalidades}

\begin{itemize}
    \item \textbf{Estadísticas:} dashboards de métricas para profesional y administrador
    (\ruta{/api/admin/metrics/overview}, \texttt{growth}, \texttt{roles},
    \texttt{top-skills}, \texttt{health}) y registro de auditoría (\ruta{/logs}).
    \item \textbf{Adaptabilidad:} internacionalización es/en/pt (i18next), modo claro/oscuro
    y responsividad final.
    \item \textbf{Notificaciones:} marcado de leídas (\texttt{PATCH .../read} y
    \texttt{read-all}).
    \item \textbf{Producción de datos:} \textit{seeders} maestros idempotentes y políticas
    de \textit{backup}/restauración.
    \item \textbf{Despliegue:} empaquetado monolítico (frontend servido por Spring Boot),
    contenedores y variables de entorno seguras.
    \item \textbf{Cierre de versión:} \textit{smoke tests} post-despliegue y regresión de la
    suite completa de sprints previos.
\end{itemize}

\section{Reporte de Anomalías y Bugs (Bug Reports)}

\subsection{Registro Detallado de Defectos}
\label{sec:s5-registro}

\begin{FichaBug}{BUG-S5-001}{Las métricas de administración eran accesibles por cualquier usuario}
\bugMeta{\sevCrit}{P1}{\texttt{admin} · control de acceso}{\resCerrado}
\bugCampo{Entorno}{Backend perfil \texttt{prod}; staging.}
\bugBloque{Pasos para reproducir}{1. Autenticarse como \texttt{PROFESSIONAL}. 2. Llamar a
\comando{GET /api/admin/metrics/overview}.}
\bugCampo{Resultado esperado}{\texttt{403 Forbidden}.}
\bugCampo{Resultado real}{\texttt{200}: métricas globales expuestas a un usuario sin rol
administrativo.}
\bugBloque{Evidencia técnica}{\texttt{GET /api/admin/metrics/overview (role=PROFESSIONAL)
$\rightarrow$ 200; sin anotación de autorización a nivel de método.}}
\bugBloque{Análisis de causa raíz (RCA)}{El controlador de administración carecía de
restricción de rol. \textbf{Corrección:} \texttt{@PreAuthorize("hasRole('ADMIN')")} en los
endpoints de administración y prueba de seguridad que cubre la matriz de roles. Verificado
por \comando{TC-S5-021} y \comando{TC-S5-034}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-002}{Los registros de auditoría exponían tokens JWT}
\bugMeta{\sevCrit}{P1}{\texttt{admin} · logs}{\resCerrado}
\bugCampo{Entorno}{Backend perfil \texttt{prod}.}
\bugBloque{Pasos para reproducir}{1. Generar eventos de login. 2. Consultar
\comando{GET /api/admin/logs}. 3. Inspeccionar el contenido.}
\bugCampo{Resultado esperado}{Eventos sin secretos.}
\bugCampo{Resultado real}{Algunos eventos incluían el \texttt{Authorization: Bearer ey...}
completo.}
\bugBloque{Análisis de causa raíz (RCA)}{El interceptor de auditoría registraba las
cabeceras sin filtrar. \textbf{Corrección:} lista de cabeceras a enmascarar
(\texttt{Authorization}, \texttt{Cookie}) y redacción de patrones de token. Verificado por
\comando{TC-S5-020} y \comando{TC-S5-035}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-007}{Clave de servicio incrustada en el bundle del frontend}
\bugMeta{\sevCrit}{P1}{\texttt{frontend} · build}{\resCerrado}
\bugCampo{Entorno}{Frontend Vite; build de producción.}
\bugBloque{Pasos para reproducir}{1. Generar el build. 2. Buscar la clave de servicio en
\texttt{dist/assets}.}
\bugCampo{Resultado esperado}{Solo claves públicas (anon).}
\bugCampo{Resultado real}{La clave de servicio (privilegiada) quedaba incrustada en el
bundle.}
\bugBloque{Análisis de causa raíz (RCA)}{La variable se expuso con el prefijo
\texttt{VITE\_} que Vite inyecta en el cliente. \textbf{Corrección:} se retiró el prefijo,
la clave de servicio se usa solo en el backend y se rotó la clave comprometida. Verificado
por \comando{TC-S5-036}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-005}{La métrica de crecimiento producía división por cero}
\bugMeta{\sevAlt}{P2}{\texttt{admin} · métricas}{\resCerrado}
\bugCampo{Entorno}{Backend perfil \texttt{prod}.}
\bugBloque{Pasos para reproducir}{1. Consultar growth cuando el periodo anterior es 0. 2.
Observar el valor.}
\bugCampo{Resultado esperado}{Valor definido (p. ej. 100\% o "n/d").}
\bugCampo{Resultado real}{\texttt{Infinity}/\texttt{NaN} en la respuesta JSON.}
\bugBloque{Análisis de causa raíz (RCA)}{División sin proteger el denominador cero.
\textbf{Corrección:} caso especial para denominador 0 que devuelve un valor acotado.
Verificado por \comando{TC-S5-001}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-008}{Consulta de crecimiento sin índice provoca full scan}
\bugMeta{\sevAlt}{P2}{\texttt{db} · rendimiento}{\resCerrado}
\bugCampo{Entorno}{Backend perfil \texttt{prod}; 10.000 usuarios.}
\bugBloque{Pasos para reproducir}{1. Cargar el dashboard a escala. 2. Medir la latencia y el
plan SQL.}
\bugCampo{Resultado esperado}{Latencia $<$ 1,5 s.}
\bugCampo{Resultado real}{3,4 s con \textit{full scan} sobre la tabla de usuarios.}
\bugBloque{Análisis de causa raíz (RCA)}{Faltaba un índice en \texttt{created\_at} usado por
la agregación temporal. \textbf{Corrección:} migración que añade el índice; latencia final
$\approx$ 420 ms. Verificado por \comando{TC-S5-038}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-003}{Claves de traducción faltantes en portugués}
\bugMeta{\sevMed}{P3}{\texttt{frontend} · i18n}{\resCerrado}
\bugCampo{Entorno}{Frontend React 18 / i18next.}
\bugBloque{Pasos para reproducir}{1. Cambiar el idioma a portugués. 2. Navegar por las
vistas.}
\bugCampo{Resultado esperado}{Toda la UI traducida.}
\bugCampo{Resultado real}{Algunas vistas mostraban la clave cruda (p. ej.
\texttt{portfolio.publish}).}
\bugBloque{Análisis de causa raíz (RCA)}{Recursos de \texttt{pt} desactualizados respecto a
\texttt{es}. \textbf{Corrección:} se completaron las claves y se añadió una verificación de
cobertura i18n en CI. Verificado por \comando{TC-S5-039}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-004}{El modo oscuro no persistía tras recargar}
\bugMeta{\sevMed}{P3}{\texttt{frontend} · tema}{\resCerrado}
\bugCampo{Entorno}{Frontend React 18.}
\bugBloque{Pasos para reproducir}{1. Activar modo oscuro. 2. Recargar la página.}
\bugCampo{Resultado esperado}{El tema oscuro permanece.}
\bugCampo{Resultado real}{Volvía a claro al recargar.}
\bugBloque{Análisis de causa raíz (RCA)}{El tema no se leía de \texttt{localStorage} al
inicializar. \textbf{Corrección:} hidratación del tema al montar la aplicación. Verificado
por \comando{TC-S5-008}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-006}{El seeder duplicaba el catálogo al re-ejecutarse}
\bugMeta{\sevMed}{P3}{\texttt{db} · seeders}{\resCerrado}
\bugCampo{Entorno}{Backend perfil \texttt{prod}; PostgreSQL (Testcontainers).}
\bugBloque{Pasos para reproducir}{1. Ejecutar el seeder dos veces. 2. Contar el catálogo.}
\bugCampo{Resultado esperado}{Sin duplicados.}
\bugCampo{Resultado real}{El catálogo se duplicaba.}
\bugBloque{Análisis de causa raíz (RCA)}{El seeder hacía \texttt{INSERT} sin verificar
existencia. \textbf{Corrección:} \texttt{INSERT ... ON CONFLICT DO NOTHING} sobre clave
natural. Verificado por \comando{TC-S5-009}.}
\end{FichaBug}

\begin{FichaBug}{BUG-S5-009}{"Marcar todas como leídas" no actualiza el contador}
\bugMeta{\sevBaj}{P4}{\texttt{frontend} · notificaciones}{\resCerrado}
\bugCampo{Entorno}{Frontend React 18.}
\bugBloque{Pasos para reproducir}{1. Tener notificaciones sin leer. 2. Pulsar "marcar todas
como leídas".}
\bugCampo{Resultado esperado}{El contador del icono baja a 0.}
\bugCampo{Resultado real}{El backend marcaba leídas, pero el contador del icono persistía.}
\bugBloque{Análisis de causa raíz (RCA)}{El estado del contador no se invalidaba tras la
acción. \textbf{Corrección:} se refresca el contador tras \comando{read-all}. Verificado por
\comando{TC-S5-023}.}
\end{FichaBug}
